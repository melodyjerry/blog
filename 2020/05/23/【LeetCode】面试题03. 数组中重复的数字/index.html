class Solution:
    def findRepeatNumber(self, nums: List[int]) -> int:
        def count(start,end):
            c,i=0,0
            while i<len(nums):
                if start<=nums[i]<=end:
                    c+=1
                i+=1 
            return c
        left,right = 0,len(nums)-1
        if right<1:return -1
        while left+1<right:
            mid = (left+right)//2
            if count(left,mid)>mid-left+1:
                right = mid
            else:
                left = mid
        if count(nums[left],nums[left])>1:return nums[left]
        if count(nums[right],nums[right])>1:return nums[right]
        return -1
